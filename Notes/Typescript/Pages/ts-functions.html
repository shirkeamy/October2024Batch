}

  if(userRole === Roles.Admin){
    console.log("You are Admin");
  }else if (userRole === Roles.Viewer){
    console.log("You are Viewer");

  }
}

function getNumber(userInput: number){
  if(userInput > 0){
   console.log(`${userInput} is positive number`);
  }else if (userInput < 0){
   console.log(`${userInput} is negative number`);
  }else {
   console.log(`${userInput} is zero`);
  }
}

getNumber(1);
